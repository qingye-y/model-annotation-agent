## 一、机审中心状态同步修复

### 1.1 问题背景

机审中心存在两个状态同步问题：

| 问题 | 现象 | 影响 |
|------|------|------|
| 弹窗关闭不刷新 | 点击"获取线上数据"后，在完成前关闭弹窗，页面列表不自动更新 | 用户看不到任务执行状态 |
| 状态显示不一致 | 顶部横幅显示"进行中"，但批次列表显示"失败" | 状态混乱，误导用户 |

### 1.2 根因分析

| 问题 | 根因 | 位置 |
|------|------|------|
| 弹窗关闭不刷新 | `closeGetDataModal` 仅隐藏弹窗，未调用刷新函数 | model_task.html:1902 |
| 无自动轮询 | `checkRunningTasks` 仅在 `init()` 调用一次，无定时器 | model_task.html:515 |
| pending状态显示为失败 | `FetchLog.status` 默认值 `'completed'`，但 `renderBatchTable` 缺少 `pending/initializing/空值` 的状态映射 | model_task.html:787 |

### 1.3 修复方案

#### 修复1：关闭弹窗时刷新列表

```javascript
// templates/model_task.html 第1931行
window.closeGetDataModal = function() {
  document.getElementById('getDataModal').classList.remove('active');
  // 关闭弹窗后刷新批次列表，确保新任务出现
  loadBatches();
  // 关闭时若有待运行任务，触发一次检查（会在下一轮自动轮询中继续）
  checkRunningTasks();
};
```

## 二、违规原因饼图动态分组展示

### 2.1 问题背景

| 需求 | 原实现 | 目标 |
|------|--------|------|
| 动态分组 | 固定 Top 10 | 占比 ≥ 2% 独立列出，< 2% 归入"其他" |
| 展开功能 | 无 | "其他"可点击展开查看明细 |
| 中心总数 | 需验证 | 应等于 non_compliant_count |

### 2.2 后端修改

**文件**：`blueprints/dashboard.py` 第194-240行

```python
# 计算违规原因分组（动态分组）
# 规则：占比 >= 2% 的类型单独列出；< 2% 全部归入"其他"
top_violation_reasons = []
all_reasons_detail = []   # 全量明细，用于"其他"展开

if violation_reason_map:
    total_violations = sum(violation_reason_map.values())
    sorted_reasons = sorted(violation_reason_map.items(), key=lambda x: x[1], reverse=True)

    # 全量明细（按次数降序）
    for tag, count in sorted_reasons:
        percentage = round(count / total_violations * 100, 1) if total_violations > 0 else 0
        all_reasons_detail.append({'name': tag, 'value': count, 'percentage': percentage})

    # 分组：>=2% 独立列出，<2% 归入"其他"
    standalone = []
    other_total = 0
    for tag, count in sorted_reasons:
        percentage = round(count / total_violations * 100, 1) if total_violations > 0 else 0
        if percentage >= 2.0:
            standalone.append({'name': tag, 'value': count, 'percentage': percentage})
        else:
            other_total += count

    # 如果有"其他"，加入分组末尾
    if other_total > 0:
        other_pct = round(other_total / total_violations * 100, 1) if total_violations > 0 else 0
        standalone.append({'name': '其他', 'value': other_total, 'percentage': other_pct})

    top_violation_reasons = standalone

# 返回字段新增 all_reasons_detail
return jsonify({
    # ... 其他字段 ...
    'top_violation_reasons': top_violation_reasons,
    'all_reasons_detail': all_reasons_detail,  # 全量明细，供"其他"展开
    # ...
})
```

function updateReasonChart(reasons, nonCompliantCount, allReasonsDetail) {
  // 保存全量明细，供"其他"展开使用
  window._allReasonsDetail = allReasonsDetail || [];
  window._currentReasonData = reasons || [];

  // 计算总违规次数（优先用 non_compliant_count）
  var totalViolations = nonCompliantCount || 0;

  // 更新副标题：共 N 例违规，X 类主要类型（含其他 N 例）
  var subtitleEl = document.getElementById('reasonChartSubtitle');
  if (subtitleEl) {
    if (totalViolations > 0) {
      var standaloneCount = (reasons || []).filter(function(d) { return d.name !== '其他'; }).length;
      var otherItem = (reasons || []).find(function(d) { return d.name === '其他'; });
      var otherInfo = otherItem ? '（含其他 ' + (otherItem.value || 0).toLocaleString() + ' 例）' : '';
      subtitleEl.textContent = '共 ' + totalViolations.toLocaleString() + ' 例违规，' + standaloneCount + ' 类主要类型' + otherInfo;
    }
  }
  // ...
}

#### 修改3 & 4：`renderOtherDetail` 和 `toggleOtherDetail` 函数

**文件**：`templates/dashboard.html` 第1256行和第1299行

```javascript
// "其他"明细展开状态
var _otherDetailExpanded = false;

function renderOtherDetail(otherItems, total) {
  var detailEl = document.getElementById('reasonOtherDetail');
  if (!detailEl) return;

  var detailData = [];
  if (window._allReasonsDetail && window._allReasonsDetail.length > 0) {
    var standaloneNames = {};
    var displayNames = window._currentReasonData || [];
    displayNames.forEach(function(d) {
      if (d.name !== '其他') standaloneNames[d.name] = true;
    });
    // 过滤出被归入"其他"的所有类型
    detailData = window._allReasonsDetail.filter(function(d) {
      return !standaloneNames[d.name];
    });
  }

  if (detailData.length === 0) {
    detailEl.style.display = 'none';
    return;
  }

  // 渲染明细列表
  var rows = detailData.map(function(d) {
    var pct = d.percentage !== undefined ? d.percentage : (d.value / total * 100);
    return '<div style="display:flex;justify-content:space-between;padding:4px 0;border-bottom:1px solid #f0f0f0;font-size:12px;">' +
      '<span>' + d.name + '</span>' +
      '<span style="color:#6b7280;">' + d.value.toLocaleString() + ' 例（' + pct.toFixed(1) + '%）</span>' +
      '</div>';
  }).join('');

  detailEl.innerHTML =
    '<div style="margin-top:8px;padding:10px;background:#f9fafb;border-radius:8px;border:1px solid #e5e7eb;">' +
    '<div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:8px;">' +
    '<span style="font-size:12px;font-weight:600;color:#374151;">其他（&lt;2%）明细（' + detailData.length + ' 类，共 ' + otherItem.value.toLocaleString() + ' 例）</span>' +
    '<span style="font-size:11px;color:#6b7280;">点击图例"其他"关闭</span>' +
    '</div>' +
    '<div style="max-height:200px;overflow-y:auto;">' + rows + '</div>' +
    '</div>';

  detailEl.style.display = _otherDetailExpanded ? 'block' : 'none';
}

function toggleOtherDetail() {
  _otherDetailExpanded = !_otherDetailExpanded;
  var detailEl = document.getElementById('reasonOtherDetail');
  if (detailEl) detailEl.style.display = _otherDetailExpanded ? 'block' : 'none';
}
```
